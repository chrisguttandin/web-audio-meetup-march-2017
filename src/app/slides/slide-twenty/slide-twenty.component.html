const downmixContext = new OfflineAudioContext( 1, audioBuffer.length, audioBuffer.sampleRate ); const bufferSource = new AudioBufferSourceNode(downmixContext, { buffer: audioBuffer }); bufferSource.start(0); bufferSource.connect(downmixContext.destination); downmixContext.startRendering(); // will resolve with the downmixed audioBuffer
